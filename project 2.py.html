def squared_sum_table(x, y):
    # create empty list
    nested_list_out = []
    for i in range(x):
        # create empty list that is the cells of the first list
        nested_list_in = []
        for j in range(y):
            value = (i+j) ** 2
            # inserting the values in to the inside list = cell of outside list
            nested_list_in.append(value)
        # inserting the inside list in to the outside list
        nested_list_out.append(nested_list_in)
    return nested_list_out

def avg_list(list1, list2):
    # create the empty list
    avg_list = []
    # check the length of the strings
    if len(list1) > len(list2):
        for i in range(len(list2)):
            # add the average values
            avg_list.append((list1[i] + list2[i]) / 2)
        for i in list1[len(list2):]:
            # add the rest of the values from the longer list
            avg_list.append(i)
    elif len(list2) > len(list1):
        for i in range(len(list1)):
            avg_list.append((list1[i] + list2[i]) / 2)
        for i in list1[len(list1):]:
            avg_list.append(i)
    else:
        for i in range(len(list1)):
            avg_list.append((list1[i] + list2[i]) / 2)
    return avg_list

def shifted_string(str1, str2):
    # create temporary string
    str_temp = []
    for i in range(len(str1)):
        # changing the order of the word
        str_temp.append(str1[i:] + str1[:i])
    # check if the new order is identical to the order that the user gave
    if str2 in str_temp:
        return "yes"
    return "no"
